def initialize(actual, matcher_1, matcher_2)
  @actual        = actual
  @matcher_1     = matcher_1
  @matcher_2     = matcher_2
  @match_results = Hash.new { |h, matcher| h[matcher] = matcher.matches?(@actual) }
end
